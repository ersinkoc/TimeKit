Quick Start

import { createTime } from '@oxog/timekit';

// Create a time object
const now = createTime();

// Format dates
now.format('YYYY-MM-DD HH:mm:ss'); // '2024-01-15 10:30:45'

// Add time
const nextWeek = now.add(7, 'days');

// Work with timezones
const tokyoTime = now.tz('Asia/Tokyo');

// Human-readable output
now.from(createTime('2024-01-01')); // '2 weeks ago'
